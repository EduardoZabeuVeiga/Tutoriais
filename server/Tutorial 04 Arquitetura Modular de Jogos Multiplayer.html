const fs = require('fs');
const path = require('path');

class GameConfigManager {
    constructor() {
        this.config = null;
        this.loadConfig();
    }

    // 📂 Carregar configurações do arquivo JSON
    loadConfig() {
        try {
            const configPath = path.join(__dirname, '..', 'config', 'games.json');
            const configData = fs.readFileSync(configPath, 'utf8');
            this.config = JSON.parse(configData);
            console.log('✅ Configurações dos jogos carregadas com sucesso');
        } catch (error) {
            console.error('❌ Erro ao carregar configurações dos jogos:', error.message);
            this.config = {};
        }
    }

    // 🎮 Obter todos os jogos habilitados ordenados
    getEnabledGames() {
        return Object.values(this.config.games)
            .filter(game => game.enabled)
            .sort((a, b) => (a.order || 999) - (b.order || 999));
    }

    // 🎯 Obter configuração de um jogo específico
    getGameConfig(gameId) {
        return this.config.games[gameId] || null;
    }

    // 👥 Obter máximo de jogadores para um tipo de sala
    getMaxPlayers(gameType) {
        const gameConfig = this.getGameConfig(gameType);
        return gameConfig ? gameConfig.maxPlayers : 2;
    }

    // 🛣️ Obter rota de um jogo
    getGameRoute(gameId) {
        const gameConfig = this.getGameConfig(gameId);
        return gameConfig ? gameConfig.route : '/';
    }

    // 📄 Obter arquivo index do jogo
    getGameIndexFile(gameId) {
        const gameConfig = this.getGameConfig(gameId);
        if (!gameConfig) return null;
        
        return path.join(__dirname, '..', gameConfig.folder, gameConfig.indexFile);
    }

    // 📋 Gerar opções para dropdown do lobby
    generateLobbyOptions() {
        const games = this.getEnabledGames();
        return games.map(game => ({
            value: game.id,
            text: `${game.displayName} (${this.getPlayerCountText(game.maxPlayers)})`,
            maxPlayers: game.maxPlayers,
            route: game.route,
            id: game.id,
            icon: game.icon
        }));
    }

    // 🏷️ Gerar mapeamento de nomes para exibição
    generateRoomDisplayNames() {
        const games = this.getEnabledGames();
        const displayNames = {};
        
        games.forEach(game => {
            displayNames[game.id] = game.name.toUpperCase();
        });
        
        return displayNames;
    }

    // 🎨 Gerar mapeamento de ícones dinamicamente
    generateTypeIcons() {
        const games = this.getEnabledGames();
        const icons = {};
        
        games.forEach(game => {
            icons[game.id] = game.icon || '❓';
        });
        
        return icons;
    }

    // 📊 Texto descritivo da quantidade de jogadores
    getPlayerCountText(maxPlayers) {
        if (maxPlayers >= 999) {
            return "infinitos jogadores";
        } else if (maxPlayers === 1) {
            return "1 jogador";
        } else {
            return `${maxPlayers} jogadores`;
        }
    }
}

module.exports = GameConfigManager;

💡 Poder do GameConfigManager: Ele lê o JSON uma vez e fornece métodos inteligentes para toda a aplicação!

🎮 Gerenciador de Sistemas de Jogos

Agora vamos criar um sistema que carrega automaticamente os jogos baseado na configuração. Crie o arquivo GameSystemManager.js na pasta utils :

const fs = require('fs');
const path = require('path');

class GameSystemManager {
    constructor(io, gameConfig) {
        this.io = io;
        this.gameConfig = gameConfig;
        this.gameSystems = new Map();
        this.loadGameSystems();
    }

    // 🎮 Carregar dinamicamente todos os sistemas de jogos baseado no games.json
    loadGameSystems() {
        const enabledGames = this.gameConfig.getEnabledGames();
        
        enabledGames.forEach(game => {
            try {
                // Pular jogos que não têm lógica específica (como chat básico)
                if (game.id === 'chat') {
                    console.log(`⏭️ Pulando ${game.name} - usa sistema básico de chat`);
                    return;
                }
                
                // Usar configuração do games.json para encontrar o arquivo
                const logicPath = path.join(__dirname, '..', game.folder, game.logicFile);
                
                if (fs.existsSync(logicPath)) {
                    const GameClass = require(logicPath);
                    const gameInstance = new GameClass(this.io);
                    
                    this.gameSystems.set(game.id, {
                        instance: gameInstance,
                        config: game
                    });
                    
                    console.log(`✅ Sistema carregado: ${game.displayName} → ${game.id}`);
                } else {
                    console.log(`⚠️ Arquivo ${game.logicFile} não encontrado em ${game.folder}`);
                }
            } catch (error) {
                console.error(`❌ Erro ao carregar ${game.displayName}:`, error.message);
            }
        });
        
        console.log(`🎮 Sistemas de jogos carregados: ${this.gameSystems.size}/${enabledGames.length}`);
    }

    // 👤 Adicionar jogador dinamicamente
    addPlayer(roomType, roomId, username) {
        const gameSystem = this.gameSystems.get(roomType);
        if (gameSystem && gameSystem.instance.adicionarJogador) {
            console.log(`🎮 Adicionando ${username} ao jogo ${roomType}`);
            gameSystem.instance.adicionarJogador(roomId, username);
            return true;
        }
        return false;
    }

    // 👋 Remover jogador dinamicamente
    removePlayer(roomType, roomId, username) {
        const gameSystem = this.gameSystems.get(roomType);
        if (gameSystem && gameSystem.instance.removerJogador) {
            console.log(`🎮 Removendo ${username} do jogo ${roomType}`);
            gameSystem.instance.removerJogador(roomId, username);
            return true;
        }
        return false;
    }

    // 🎯 Processar jogada dinamicamente
    processGameMove(roomType, gameData) {
        const gameSystem = this.gameSystems.get(roomType);
        if (gameSystem) {
            // Tentar diferentes métodos comuns
            if (gameSystem.instance.rolarDados) {
                // Para o jogo de dados, mapear os parâmetros corretamente
                const tipoDado = gameData.diceType || gameData.tipoDado;
                const quantidade = gameData.quantity || gameData.quantidade || 1;
                gameSystem.instance.rolarDados(gameData.roomId, gameData.username, tipoDado, quantidade);
                return true;
            }
            if (gameSystem.instance.processMove) {
                gameSystem.instance.processMove(gameData);
                return true;
            }
            if (gameSystem.instance.handleMove) {
                gameSystem.instance.handleMove(gameData);
                return true;
            }
        }
        return false;
    }

    // 📋 Listar sistemas carregados
    getLoadedSystems() {
        return Array.from(this.gameSystems.keys());
    }

    // 🔍 Verificar se jogo existe
    hasGameSystem(roomType) {
        return this.gameSystems.has(roomType);
    }
}

module.exports = GameSystemManager;

🚀 GameSystemManager: Sistema dinâmico que carrega automaticamente todos os jogos baseado no games.json!

        // Usar o sistema dinâmico para processar a jogada
        const room = rooms.get(socket.roomId);
        if (room) {
            const processed = gameSystemManager.processGameMove(room.type, gameData);
            if (!processed) {
                console.log(`⚠️ Nenhum sistema encontrado para processar jogada do tipo: ${room.type}`);
            }
        }
    });
});

class Chat {
    constructor(io) {
        this.io = io;
        this.chatStates = new Map();
    }

    // Inicializar chat para uma sala
    initializeChat(roomId) {
        if (!this.chatStates.has(roomId)) {
            this.chatStates.set(roomId, {
                messages: [],
                users: new Set()
            });
            console.log(`💬 Inicializando Chat para sala: ${roomId}`);
        }
        return this.chatStates.get(roomId);
    }

    // Adicionar usuário ao chat
    addUser(roomId, username) {
        const chatState = this.initializeChat(roomId);
        chatState.users.add(username);
        
        // Notificar outros usuários
        this.io.to(roomId).emit('user_joined', {
            username: username,
            message: `${username} entrou na sala`,
            timestamp: new Date()
        });
        
        console.log(`👤 Usuário ${username} entrou no chat da sala: ${roomId}`);
    }

    // Processar mensagem do chat
    handleMessage(data) {
        const { roomId, username, message } = data;
        
        console.log(`💬 Mensagem recebida: ${username} em ${roomId}: ${message}`);
        
        const chatState = this.initializeChat(roomId);
        
        const messageData = {
            id: Date.now() + Math.random(),
            username: username,
            message: message,
            timestamp: new Date()
        };
        
        // Adicionar mensagem ao histórico
        chatState.messages.push(messageData);
        
        // Manter apenas as últimas 100 mensagens
        if (chatState.messages.length > 100) {
            chatState.messages = chatState.messages.slice(-100);
        }
        
        // Enviar mensagem para todos na sala
        this.io.to(roomId).emit('chat_message', messageData);
        
        return messageData;
    }

    // Remover usuário do chat  
    removeUser(roomId, username) {
        const chatState = this.chatStates.get(roomId);
        if (chatState) {
            chatState.users.delete(username);
            
            // Notificar outros usuários
            this.io.to(roomId).emit('user_left', {
                username: username,
                message: `${username} saiu da sala`,
                timestamp: new Date()
            });
            
            console.log(`👋 Usuário ${username} saiu do chat da sala: ${roomId}`);
            
            // Limpar estado se não há mais usuários
            if (chatState.users.size === 0) {
                this.cleanupChat(roomId);
            }
        }
    }

    // Limpar estado do chat
    cleanupChat(roomId) {
        if (this.chatStates.has(roomId)) {
            this.chatStates.delete(roomId);
            console.log(`🗑️ Removendo estado do Chat para sala: ${roomId}`);
        }
    }
}

module.exports = Chat;

❌ Problema: Usuários não se veem nas salas

  • Lista "Usuários Online" aparece vazia ou só mostra você
  • Mensagens não chegam para outros usuários
  • Múltiplos usuários na mesma sala não se veem

  1. ✅ Ler parâmetro room da URL
  2. ✅ Emitir evento join_room
  3. ✅ Implementar handlers room_joined, user_joined, user_left
  4. ✅ Adicionar tratamento de erros

❌ Problema: Botão "Sair" desloga completamente

Sintomas: Ao clicar "Sair" na sala, volta para tela de login

🔧 Solução: Atualizar função logout() conforme seção 6.2

❌ Problema: Erro "Sala não encontrada"

Possíveis Causas:

  • Parâmetro room não está sendo passado na URL
  • Sala foi excluída enquanto usuário tentava entrar
  • Problema na função joinRoom() do lobby

🔧 Verificações:

  1. Console do navegador para erros JavaScript
  2. URL deve ter formato: /chat?room=12345
  3. Logs do servidor para sala sendo criada/excluída

❌ Problema: Configurações não carregam

Sintomas: Lobby vazio ou dropdown sem opções

🔧 Soluções:

  • Verificar se config/games.json existe
  • Verificar se chat tem "enabled": true
  • Testar URL: http://localhost:3000/api/games/config
