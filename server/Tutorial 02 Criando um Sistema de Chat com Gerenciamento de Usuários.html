Tutorial 02: Criando um Sistema de Chat com Gerenciamento de Usuários

Pré-requisitos:
  • Tutorial 01 completado (servidor básico funcionando)
  • Node.js e dependências já instaladas
  • Projeto aberto no VS Code

1. Criando o Servidor de Chat Avançado

Vamos criar um novo arquivo para o servidor de chat. Crie um arquivo chamado server-chat.js na raiz do projeto com o seguinte código:

const express = require('express');
const app = express();
const http = require('http').createServer(app);
const io = require('socket.io')(http);

// Rota principal redireciona para o chat
app.get('/', (req, res) => {
    res.sendFile(__dirname + '/public/chat.html');
});

// Servir arquivos estáticos da pasta 'public'
app.use(express.static('public'));

// Lista de usuários conectados
const usuarios = new Set();

// Configuração do Socket.IO com sistema de usuários
io.on('connection', (socket) => {
    let username = '';

    // Quando um novo usuário entra
    socket.on('novo usuario', (novoUsername) => {
        username = novoUsername;
        usuarios.add(username);
        
        // Avisar a todos que um novo usuário entrou
        io.emit('system message', `${username} entrou no chat`);
        
        // Atualizar lista de usuários para todos
        io.emit('users list', Array.from(usuarios));
    });

    // Quando um usuário envia uma mensagem
    socket.on('chat message', (data) => {
        io.emit('chat message', {
            username: data.username,
            message: data.message
        });
    });

    // Quando um usuário desconecta
    socket.on('disconnect', () => {
        if (username) {
            usuarios.delete(username);
            io.emit('system message', `${username} saiu do chat`);
            io.emit('users list', Array.from(usuarios));
        }
    });
});

// Iniciando o servidor na porta 3000
const PORT = process.env.PORT || 3000;
http.listen(PORT, () => {
    console.log(`Servidor rodando na porta ${PORT}`);
});

2. Editando o package.json

Vamos atualizar o arquivo package.json para usar o novo servidor de chat. Modifique a seção "scripts" para:

"scripts": {
    "start": "node server-chat.js",
    "test": "echo \"Error: no test specified\" && exit 1"
},

<body>
    <div id="chat-container">

                <input type="text" id="username" placeholder="Seu nome" />
                <button id="enter-btn" onclick="entrarNoChat()">Entrar no Chat</button>
                <div id="message-input">
                    <input type="text" id="message" placeholder="Digite sua mensagem" />
                    <button onclick="enviarMensagem()">Enviar</button>

            <h3>Usuários Online</h3>
            <ul id="users"></ul>

    <script>
        const socket = io();
        let meuUsername = '';

        // Elementos do DOM
        const messagesDiv = document.getElementById('messages');
        const usernameInput = document.getElementById('username');
        const messageInput = document.getElementById('message');
        const messageControls = document.getElementById('message-input');
        const usersList = document.getElementById('users');
        const enterBtn = document.getElementById('enter-btn');

        function entrarNoChat() {
            const username = usernameInput.value.trim();
            if (username) {
                meuUsername = username;
                socket.emit('novo usuario', username);
                usernameInput.disabled = true;
                enterBtn.style.display = 'none';
                messageControls.style.display = 'flex';
                messageInput.focus();
            }
        }

        function enviarMensagem() {
            const message = messageInput.value.trim();
            if (message) {
                socket.emit('chat message', {
                    username: meuUsername,
                    message: message
                });
                messageInput.value = '';
            }
        }

        // Pressionar Enter para enviar mensagem
        messageInput.addEventListener('keypress', (e) => {
            if (e.key === 'Enter') {
                enviarMensagem();
            }
        });

        // Pressionar Enter para entrar no chat
        usernameInput.addEventListener('keypress', (e) => {
            if (e.key === 'Enter') {
                entrarNoChat();
            }
        });

        // Receber mensagens
        socket.on('chat message', (data) => {
            const messageElement = document.createElement('div');
            messageElement.className = 'message user-message';
            messageElement.textContent = `${data.username}: ${data.message}`;
            messagesDiv.appendChild(messageElement);
            messagesDiv.scrollTop = messagesDiv.scrollHeight;
        });

        // Atualizar lista de usuários
        socket.on('users list', (users) => {
            usersList.innerHTML = '';
            users.forEach(user => {
                const li = document.createElement('li');
                li.textContent = user;
                usersList.appendChild(li);
            });
        });

        // Mensagens do sistema
        socket.on('system message', (message) => {
            const messageElement = document.createElement('div');
            messageElement.className = 'message system-message';
            messageElement.textContent = message;
            messagesDiv.appendChild(messageElement);
            messagesDiv.scrollTop = messagesDiv.scrollHeight;
        });

        // Status de conexão
        socket.on('connect', () => {
            console.log('Conectado ao servidor');
        });

        socket.on('disconnect', () => {
            console.log('Desconectado do servidor');
        });
    </script>
</body>
</html>

4. Executando o Servidor de Chat

Agora vamos iniciar o servidor com as novas funcionalidades:

npm start

Seu servidor de chat estará rodando em http://localhost:3000 Abrir

📍 Importante: O servidor foi configurado para servir automaticamente a interface do chat quando você acessar http://localhost:3000

5. Testando o Chat

Para testar o sistema de chat completo:

  1. Acesse http://localhost:3000 Abrir (vai direto para o chat!)
  2. Digite seu nome e clique em "Entrar no Chat"
  3. Você verá a mensagem de sistema informando que entrou
  4. Seu nome aparecerá na lista de usuários online

Como Testar com Múltiplos Usuários

Para simular múltiplos usuários no chat:

  1. Abra uma janela normal do navegador e acesse http://localhost:3000
  2. Entre com um nome (ex: "João")
  3. Abra uma janela anônima (Ctrl + Shift + N)
  4. Acesse o mesmo link na janela anônima
  5. Entre com outro nome (ex: "Maria")
  6. Agora você pode conversar entre as duas janelas!

Você verá que:

  • Mensagens de sistema aparecem quando usuários entram/saem
  • A lista de usuários online é atualizada em tempo real
  • Mensagens aparecem instantaneamente para todos
  • O scroll é automático nas mensagens

6. Funcionalidades Implementadas

O sistema de chat agora possui:

  • Gerenciamento de usuários: Cada usuário tem um nome único
  • Lista de usuários online: Mostra quem está conectado em tempo real
  • Mensagens de sistema: Notifica quando usuários entram ou saem
  • Chat em tempo real: Mensagens instantâneas entre todos os usuários
  • Interface intuitiva: Design limpo e fácil de usar
  • Scroll automático: Sempre mostra as mensagens mais recentes
  • Suporte a Enter: Pressione Enter para enviar mensagens
  • Validação: Campos não podem estar vazios
