2. Atualizando o Servidor para Suportar Login

Vamos modificar o arquivo server-chat.js para incluir autenticação. Substitua todo o conteúdo do arquivo pelo código abaixo:

const express = require('express');
const app = express();
const http = require('http').createServer(app);
const io = require('socket.io')(http);
const fs = require('fs');
const path = require('path');

// Middleware para processar JSON
app.use(express.json());

// Carregar usuários do arquivo
let users = [];
try {
    const usersData = fs.readFileSync('users.json', 'utf8');
    users = JSON.parse(usersData);
} catch (error) {
    console.log('Erro ao carregar usuários:', error.message);
    users = [];
}

// Rota principal redireciona para o login
app.get('/', (req, res) => {
    res.sendFile(__dirname + '/public/login.html');
});

// Rota do chat (só acessível após login)
app.get('/chat', (req, res) => {
    res.sendFile(__dirname + '/public/chat.html');
});

// Rota para verificar login
app.post('/login', (req, res) => {
    const { username, password } = req.body;
    
    const user = users.find(u => u.username === username && u.password === password);
    
    if (user) {
        res.json({ 
            success: true, 
            message: 'Login realizado com sucesso!',
            user: { username: user.username, role: user.role }
        });
    } else {
        res.json({ 
            success: false, 
            message: 'Usuário ou senha incorretos!' 
        });
    }
});

// Servir arquivos estáticos da pasta 'public'
app.use(express.static('public'));

// Lista de usuários conectados
const usuarios = new Set();

// Configuração do Socket.IO com sistema de usuários
io.on('connection', (socket) => {
    let username = '';

    // Quando um novo usuário entra
    socket.on('novo usuario', (novoUsername) => {
        username = novoUsername;
        usuarios.add(username);
        
        // Avisar a todos que um novo usuário entrou
        io.emit('system message', `${username} entrou no chat`);
        
        // Atualizar lista de usuários para todos
        io.emit('users list', Array.from(usuarios));
    });

    // Quando um usuário envia uma mensagem
    socket.on('chat message', (data) => {
        io.emit('chat message', {
            username: data.username,
            message: data.message
        });
    });

    // Quando um usuário desconecta
    socket.on('disconnect', () => {
        if (username) {
            usuarios.delete(username);
            io.emit('system message', `${username} saiu do chat`);
            io.emit('users list', Array.from(usuarios));
        }
    });
});

// Iniciando o servidor na porta 3000
const PORT = process.env.PORT || 3000;
http.listen(PORT, () => {
    console.log(`Servidor rodando na porta ${PORT}`);
});

        const loginForm = document.getElementById('loginForm');
        const loginBtn = document.getElementById('loginBtn');
        const messageDiv = document.getElementById('message');
        const usernameInput = document.getElementById('username');
        const passwordInput = document.getElementById('password');

        function showMessage(text, type) {
            messageDiv.textContent = text;
            messageDiv.className = `message ${type}`;
            messageDiv.style.display = 'block';
        }

        function hideMessage() {
            messageDiv.style.display = 'none';
        }

        loginForm.addEventListener('submit', async (e) => {
            e.preventDefault();
            hideMessage();
            
            const username = usernameInput.value.trim();
            const password = passwordInput.value.trim();
            
            if (!username || !password) {
                showMessage('Por favor, preencha todos os campos!', 'error');
                return;
            }
            
            loginBtn.disabled = true;
            loginBtn.textContent = 'Entrando...';
            
            try {
                const response = await fetch('/login', {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json'
                    },
                    body: JSON.stringify({ username, password })
                });
                
                const result = await response.json();
                
                if (result.success) {
                    showMessage(result.message, 'success');
                    // Salvar dados do usuário
                    sessionStorage.setItem('user', JSON.stringify(result.user));
                    // Redirecionar para o chat após 1 segundo
                    setTimeout(() => {
                        window.location.href = '/chat';
                    }, 1000);
                } else {
                    showMessage(result.message, 'error');
                }
            } catch (error) {
                showMessage('Erro ao conectar com o servidor!', 'error');
            } finally {
                loginBtn.disabled = false;
                loginBtn.textContent = 'Entrar no Chat';
            }
        });

        // Enter para enviar
        passwordInput.addEventListener('keypress', (e) => {
            if (e.key === 'Enter') {
                loginForm.dispatchEvent(new Event('submit'));
            }
        });
    </script>
</body>

        // Verificar se o usuário está logado
        const userData = sessionStorage.getItem('user');
        if (!userData) {
            window.location.href = '/';
            throw new Error('Usuário não logado');
        }
        
        const user = JSON.parse(userData);
        const socket = io();

        const messageInput = document.getElementById('message');
        const usersList = document.getElementById('users');
        const currentUserSpan = document.getElementById('current-user');

        // Entrar automaticamente no chat com o usuário logado

        function enviarMensagem() {
            const message = messageInput.value.trim();
            if (message) {
                socket.emit('chat message', {
                    username: user.username,
                    message: message
                });
                messageInput.value = '';
            }
        }

            sessionStorage.removeItem('user');
            window.location.href = '/';

            if (e.key === 'Enter') {
                enviarMensagem();
            }
        });

        // Receber mensagens
        socket.on('chat message', (data) => {
            const messageElement = document.createElement('div');
            const isMyMessage = data.username === user.username;
            messageElement.className = `message ${isMyMessage ? 'my-message' : 'user-message'}`;
            messageElement.textContent = `${data.username}: ${data.message}`;
            messagesDiv.appendChild(messageElement);
            messagesDiv.scrollTop = messagesDiv.scrollHeight;
        });

        // Atualizar lista de usuários
        socket.on('users list', (users) => {
            usersList.innerHTML = '';
            users.forEach(username => {
                const li = document.createElement('li');
                li.textContent = username;
                if (username === user.username) {
                    li.style.backgroundColor = '#c8e6c9';
                    li.style.borderLeftColor = '#4caf50';
                }
                usersList.appendChild(li);
            });
        });

        // Mensagens do sistema
        socket.on('system message', (message) => {
            const messageElement = document.createElement('div');
            messageElement.className = 'message system-message';
            messageElement.textContent = message;
            messagesDiv.appendChild(messageElement);
            messagesDiv.scrollTop = messagesDiv.scrollHeight;
        });

        // Status de conexão
        socket.on('connect', () => {
            console.log('Conectado ao servidor');
        });

        socket.on('disconnect', () => {
            console.log('Desconectado do servidor');
        });
        
        // Focar no campo de mensagem
        messageInput.focus();

5. Testando o Sistema de Login

Agora vamos testar o sistema completo:

npm start

Acesse http://localhost:3000 Abrir

  1. Você verá a tela de login com os usuários de teste
  2. Teste com um dos usuários:
    • admin / 123456
    • joao / senha123
    • maria / maria2024
  3. Após login bem-sucedido, será redirecionado para o chat
  4. O chat mostrará seu nome de usuário no cabeçalho
  5. Suas mensagens aparecerão destacadas em verde

Como Testar com Múltiplos Usuários

Para simular múltiplos usuários logados:

  1. Faça login com um usuário em uma janela normal
  2. Abra uma janela anônima (Ctrl + Shift + N)
  3. Acesse http://localhost:3000 na janela anônima
  4. Faça login com outro usuário
  5. Agora você pode conversar entre os dois usuários!

6. Funcionalidades Implementadas

O sistema agora possui:

  • Autenticação segura: Login com usuário e senha
  • Proteção de rotas: Chat só acessível após login
  • Sessão de usuário: Dados salvos durante a navegação
  • Interface de login: Design moderno e responsivo
  • Identificação visual: Mensagens próprias destacadas
  • Sistema de logout: Sair do chat com segurança
  • Usuários de teste: Três contas pré-configuradas
  • Validação de dados: Verificação de campos obrigatórios
