1.5 Conteúdo do `renderer.js`

Crie o arquivo renderer.js com a lógica da interface:

// renderer.js - Script que roda na interface do usuário

// Aguarda o DOM carregar completamente
document.addEventListener('DOMContentLoaded', () => {
    // Exibir informações do sistema
    displaySystemInfo();
    
    // Configurar eventos dos botões
    setupEventListeners();
    
    // Inicializar contador
    let counter = 0;
    updateCounterDisplay(counter);
    
    // Funções do contador
    document.getElementById('btn-increment').addEventListener('click', () => {
        counter++;
        updateCounterDisplay(counter);
        showFeedback('Incrementado!', 'success');
    });
    
    document.getElementById('btn-decrement').addEventListener('click', () => {
        counter--;
        updateCounterDisplay(counter);
        showFeedback('Decrementado!', 'info');
    });
});

function displaySystemInfo() {
    // Mostrar versões do sistema
    document.getElementById('electron-version').textContent = process.versions.electron;
    document.getElementById('node-version').textContent = process.versions.node;
    document.getElementById('chrome-version').textContent = process.versions.chrome;
}

function setupEventListeners() {
    // Botão de notificação
    document.getElementById('btn-notification').addEventListener('click', () => {
        showNotification();
    });
    
    // Botão de diálogo
    document.getElementById('btn-dialog').addEventListener('click', () => {
        showDialog();
    });
    
    // Botão de nova janela
    document.getElementById('btn-new-window').addEventListener('click', () => {
        openNewWindow();
    });
}

function showNotification() {
    if ('Notification' in window) {
        new Notification('Olá do Electron!', {
            body: 'Esta é uma notificação do seu app desktop.',
            icon: 'https://electronjs.org/images/electron-logo.svg'
        });
        showFeedback('Notificação enviada!', 'success');
    } else {
        showFeedback('Notificações não suportadas', 'error');
    }
}

function showDialog() {
    const message = prompt('Digite uma mensagem:');
    if (message) {
        alert(`Você digitou: ${message}`);
        showFeedback('Diálogo exibido!', 'success');
    } else {
        showFeedback('Diálogo cancelado', 'info');
    }
}

function openNewWindow() {
    // Esta função seria implementada no processo principal
    // Por enquanto, apenas simula a ação
    showFeedback('Nova janela seria aberta (funcionalidade do processo principal)', 'info');
}

function updateCounterDisplay(value) {
    const element = document.getElementById('counter-value');
    element.textContent = value;
    
    // Adicionar animação
    element.style.transform = 'scale(1.2)';
    setTimeout(() => {
        element.style.transform = 'scale(1)';
    }, 150);
}

function showFeedback(message, type = 'info') {
    // Criar elemento de feedback
    const feedback = document.createElement('div');
    feedback.textContent = message;
    feedback.style.cssText = `
        position: fixed;
        top: 20px;
        right: 20px;
        padding: 12px 20px;
        border-radius: 8px;
        color: white;
        font-weight: 600;
        z-index: 1000;
        animation: slideIn 0.3s ease;
    `;
    
    // Definir cor baseada no tipo
    const colors = {
        success: '#4ade80',
        error: '#ef4444',
        info: '#3b82f6',
        warning: '#f59e0b'
    };
    
    feedback.style.backgroundColor = colors[type] || colors.info;
    
    // Adicionar ao body
    document.body.appendChild(feedback);
    
    // Remover após 3 segundos
    setTimeout(() => {
        feedback.style.animation = 'slideOut 0.3s ease';
        setTimeout(() => {
            document.body.removeChild(feedback);
        }, 300);
    }, 3000);
}

// Adicionar CSS para animações
const style = document.createElement('style');
style.textContent = `
    @keyframes slideIn {
        from { transform: translateX(100%); opacity: 0; }
        to { transform: translateX(0); opacity: 1; }
    }
    
    @keyframes slideOut {
        from { transform: translateX(0); opacity: 1; }
        to { transform: translateX(100%); opacity: 0; }
    }
`;
document.head.appendChild(style);
1.6 Conteúdo Padrão do `main.js`

O arquivo main.js é o ponto de entrada do seu aplicativo. Este código padrão, com comentários, serve para a maioria dos projetos:

// main.js
const { app, BrowserWindow } = require('electron');
const path = require('path');

const createWindow = () => {
  // Cria a janela do navegador.
  const win = new BrowserWindow({
    width: 1280,
    height: 800,
    webPreferences: {
      // É uma boa prática de segurança usar um script de preload.
      // preload: path.join(__dirname, 'preload.js') 
    }
  });

  // Carrega o index.html do aplicativo.
  win.loadFile('index.html');

  // Abre o DevTools (ferramentas de desenvolvedor).
  // Remova ou comente esta linha para a versão de produção.
  // win.webContents.openDevTools();
};

// Este método será chamado quando o Electron tiver finalizado
// a inicialização e estiver pronto para criar janelas do navegador.
app.whenReady().then(() => {
  createWindow();

  app.on('activate', () => {
    // No macOS, é comum recriar uma janela no aplicativo quando o
    // ícone da dock é clicado e não há outras janelas abertas.
    if (BrowserWindow.getAllWindows().length === 0) {
      createWindow();
    }
  });
});

// Encerra o aplicativo quando todas as janelas são fechadas, exceto no macOS.
app.on('window-all-closed', () => {
  if (process.platform !== 'darwin') {
    app.quit();
  }
});

2. Configurando o `package.json`

O próximo passo é configurar o arquivo package.json. O comando npm init -y (que rodaremos a seguir) cria um arquivo básico, mas precisamos de mais informações para o Electron. Use o exemplo abaixo como seu conteúdo final.

Atenção: Copie o conteúdo abaixo e cole no seu arquivo package.json. Depois, edite os campos name, author, description, appId, productName e publisherName com os dados do seu app. 
{
  "name": "meu-app",
  "version": "1.0.0",
  "main": "main.js",
  "author": "Seu Nome",
  "description": "Descrição do seu app.",
  "scripts": {
    "start": "electron .",
    "dist": "electron-builder"
  },
  "build": {
    "appId": "com.suaempresa.meuapp",
    "productName": "Meu Aplicativo",
    "files": ["**/*"],
    "win": {
      "target": "nsis",
      "icon": "icon.ico"
    },
    "nsis": {
      "oneClick": false,
      "allowToChangeInstallationDirectory": true
    },
    "publisherName": "Sua Empresa LTDA"
  },
  "devDependencies": {
    "electron": "^31.0.0",
    "electron-builder": "^24.13.3"
  }
}

3. Instalando Dependências

Agora, com o package.json criado e preenchido, vamos instalar o Electron e o Electron Builder. Abra o terminal na pasta do seu projeto e rode os comandos:

npm init -y
npm install
O comando npm install irá ler seu package.json e instalar as dependências listadas em devDependencies (Electron e Electron Builder).

4. Executando o App em Desenvolvimento

Com tudo instalado e configurado, você já pode testar o app. Use o comando:

npm start
Este comando executa o script "start" que você definiu no package.json.

5. Como Testar a Aplicação

5.1 Testando Funcionalidades Básicas

Após executar npm start, sua aplicação deve abrir. Teste os seguintes recursos:

  • Informações do Sistema: Verifique se as versões do Electron, Node.js e Chrome estão sendo exibidas
  • Botão de Notificação: Clique para ver uma notificação do sistema
  • Contador: Use os botões + e - para incrementar/decrementar
  • Diálogo: Teste a funcionalidade de prompt e alert
5.2 Testando com Múltiplas Instâncias

Para testar como sua aplicação se comporta com múltiplas janelas/instâncias:

  1. Execute npm start em um terminal
  2. Abra outro terminal na mesma pasta do projeto
  3. Execute npm start novamente
  4. Observe se duas janelas são abertas ou se a aplicação impede múltiplas instâncias
Dica: Para controlar múltiplas instâncias, você pode usar app.requestSingleInstanceLock() no main.js para permitir apenas uma instância da aplicação por vez.
5.3 Testando em Modo de Desenvolvimento

Durante o desenvolvimento, você pode:

  • Abrir DevTools: Pressione Ctrl+Shift+I (Windows/Linux) ou Cmd+Option+I (Mac)
  • Recarregar a Página: Pressione Ctrl+R ou F5
  • Forçar Reload: Pressione Ctrl+Shift+R
Atenção: Descomente a linha win.webContents.openDevTools(); no main.js para abrir automaticamente o DevTools ao iniciar a aplicação.

6. Configurando o Ícone do Aplicativo

Antes de empacotar, é essencial configurar o ícone que representará seu aplicativo.

icon.ico
  • Nome do Arquivo: O nome icon.ico é uma convenção. O mais importante é que o nome do arquivo na pasta do projeto seja exatamente o mesmo que você especificou no `package.json` (na linha "icon": "icon.ico").
  • Localização: Coloque o arquivo .ico na raiz do projeto.
  • Formato e Especificações: Para Windows, o formato deve ser .ico. É recomendável que este arquivo contenha múltiplos tamanhos (ex: 16x16, 32x32, 256x256) para garantir sua exibição correta.

9. Assinatura Digital e Publicação

  • Adquira um certificado digital de uma autoridade certificadora (Sectigo, DigiCert).
  • Preencha appId, author, publisherName corretamente para publicação em lojas.

10. Erros Comuns e Soluções

  • Erro: "Can't open output file"
    Solução: Feche qualquer instalador do seu app que esteja aberto e tente o comando novamente.
  • Erro: Permissão de escrita em arquivos de dados
    Solução: Use app.getPath('userData') para obter uma pasta segura para salvar dados do usuário.
  • Erro: "Module not found"
    Solução: Verifique se executou npm install corretamente e se todas as dependências estão no package.json.
  • Erro: Aplicação não abre
    Solução: Verifique se o arquivo main.js está na raiz do projeto e se o campo "main" no package.json aponta para ele.

11. Dicas Avançadas

11.1 Hot Reload durante Desenvolvimento

Para facilitar o desenvolvimento, você pode instalar o electron-reload:

npm install --save-dev electron-reload

E adicionar no início do main.js:

// Adicione no início do main.js (apenas para desenvolvimento)
if (process.env.NODE_ENV === 'development') {
    require('electron-reload')(__dirname, {
        electron: require('path').join(__dirname, 'node_modules', '.bin', 'electron'),
        hardResetMethod: 'exit'
    });
}
